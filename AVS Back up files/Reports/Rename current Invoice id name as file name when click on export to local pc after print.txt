[ExtensionOf(classStr(SrsReportRunController))]
internal final class DGMA_SrsReportRunController_Extension
{
    public LabelType parmDialogCaption(LabelType _dialogCaption)

    {
        LabelType           reportDialogCaption;

        Args                    args;

        ProjInvoiceJour  ProjInvoiceJour;

        Name                  rdpName;

        reportDialogCaption = next parmDialogCaption(_dialogCaption);
        psaContractLineInvoiceDP psaContractLineInvoiceDP;

        if (this.getReportContract().parmRdpName() == 'psaContractLineInvoiceDP')

        {

            args = this.parmArgs();

            ProjInvoiceJour = args.record();

            if(ProjInvoiceJour)

            {
                reportDialogCaption = strFmt('%1', ProjInvoiceJour.ProjInvoiceId); // Set the file name
            }
        }

        return reportDialogCaption;

    }
}
