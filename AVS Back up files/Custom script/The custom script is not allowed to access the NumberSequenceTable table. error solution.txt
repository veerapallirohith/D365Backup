internal final class STE_CustomScriptForBatchInsert
{
    /// <summary>
    /// Class entry point. The system will call this method when a designated menu 
    /// is selected or when execution starts and this class is set as the startup class.
    /// </summary>
    /// <param name = "_args">The specified arguments.</param>
    public static void main(Args _args)
    {
        SysFlighting  sysFlighting,sysFlightingJob,sysFlightingAlerts,sysFlightingJobAlerts;
      
        select sysFlighting where sysFlighting.FlightName == 'AppConsistencyCustomScriptAllowBatch';
        select sysFlightingJob where sysFlightingJob.FlightName == 'AppConsistencyCustomScriptAllowBatchJob';
        select sysFlightingAlerts where sysFlightingAlerts.FlightName ==  'AppConsistencyCustomScriptAllowBatchJobAlerts';
        select sysFlightingJobAlerts where sysFlightingJobAlerts.FlightName ==  'AppConsistencyCustomScriptAllowBatchJobRecurrenceCount';
       
        ttsbegin;
        if(!sysFlighting)
        {
            sysFlighting.FlightName = 'AppConsistencyCustomScriptAllowBatch';
            sysFlighting.Enabled = NoYes::Yes;
            sysFlighting.insert();
        }
        if(!sysFlightingJob)
        {
            sysFlightingJob.FlightName = 'AppConsistencyCustomScriptAllowBatchJob';
            sysFlightingJob.Enabled = NoYes::Yes;
            sysFlightingJob.insert();
        }
        if(!sysFlightingAlerts)
        {
            sysFlightingAlerts.FlightName = 'AppConsistencyCustomScriptAllowBatchJobAlerts';
            sysFlightingAlerts.Enabled = NoYes::Yes;
            sysFlightingAlerts.insert();
        }
        if(!sysFlightingJobAlerts)
        {
            sysFlightingJobAlerts.FlightName = 'AppConsistencyCustomScriptAllowBatchJobRecurrenceCount';
            sysFlightingJobAlerts.Enabled = NoYes::Yes;
            sysFlightingJobAlerts.insert();
        }
        ttscommit;
    }

}
